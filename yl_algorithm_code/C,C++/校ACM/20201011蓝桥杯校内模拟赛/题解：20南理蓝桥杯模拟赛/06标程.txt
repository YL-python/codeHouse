#include <bits/stdc++.h>

using namespace std;

char a[1100000];

int main() {
    int n,sum=0;
    scanf("%d",&n);
    scanf("%s",a);
    for (int i=0; i<n; i++){
        if (a[i]=='1') {
            sum++;
            a[i]='0';
            if (a[i+1]=='0') a[i+1]='1';
            else a[i+1]='0';
            if (a[i+2]=='0') a[i+2]='1';
            else a[i+2]='0';
//            cout << i << " ";
//            printf("%s\n",a);
        }
    }
    cout << sum << endl;
    return 0;
}
